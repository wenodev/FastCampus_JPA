```yaml
# JPA 컨피그 설정
jpa:
	show-sql: true # sql을 보여주는 설정
	properties:
		hibernate:
			format_sql: true # 가독성을 좋게하기 위한 설정 -> 일반적으로 생각하는 SQL문 형태로 노출된다.
```
